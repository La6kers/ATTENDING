# apps/patient-portal/components/chat/ChatInterface.tsx
```typescript
import React, { useState, useRef, useEffect } from 'react';
import { useDispatch, useSelector } from 'react-redux';
import { Button, Card, TextInput } from '@attending/ui-system';
import { MessageBubble } from './MessageBubble';
import { AssessmentProgress } from './AssessmentProgress';
import { EmergencyModal } from './EmergencyModal';
import { useAssessmentState } from '../../hooks/useAssessmentState';
import { useEmergencyProtocols } from '../../hooks/useEmergencyProtocols';
import type { Message, AssessmentState } from '@attending/types';

export const ChatInterface: React.FC = () => {
  const dispatch = useDispatch();
  const messagesEndRef = useRef<HTMLDivElement>(null);
  const [inputValue, setInputValue] = useState('');
  const [isTyping, setIsTyping] = useState(false);
  
  const { messages, currentStep, progress } = useAssessmentState();
  const { checkForEmergency, showEmergencyModal } = useEmergencyProtocols();

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  };

  useEffect(() => {
    scrollToBottom();
  }, [messages]);

  const handleSendMessage = async (text: string) => {
    if (!text.trim()) return;
    
    // Add user message
    const userMessage: Message = {
      id: Date.now().toString(),
      text,
      sender: 'user',
      timestamp: new Date(),
    };
    
    dispatch({ type: 'assessment/addMessage', payload: userMessage });
    setInputValue('');
    
    // Check for emergency keywords
    const isEmergency = checkForEmergency(text);
    if (isEmergency) {
      return;
    }
    
    // Process message and get AI response
    setIsTyping(true);
    try {
      // API call to assessment service
      const response = await assessmentAPI.processMessage(text, currentStep);
      
      const aiMessage: Message = {
        id: Date.now().toString(),
        text: response.message,
        sender: 'ai',
        timestamp: new Date(),
        suggestions: response.suggestions,
      };
      
      dispatch({ type: 'assessment/addMessage', payload: aiMessage });
      dispatch({ type: 'assessment/updateProgress', payload: response.progress });
      
      if (response.nextStep) {
        dispatch({ type: 'assessment/setStep', payload: response.nextStep });
      }
    } catch (error) {
      console.error('Error processing message:', error);
    } finally {
      setIsTyping(false);
    }
  };

  const handleQuickReply = (text: string) => {
    handleSendMessage(text);
  };

  return (
    <div className="chat-interface">
      <Card className="chat-container">
        <AssessmentProgress progress={progress} currentStep={currentStep} />
        
        <div className="messages-container">
          {messages.map((message) => (
            <MessageBubble
              key={message.id}
              message={message}
              onQuickReply={handleQuickReply}
            />
          ))}
          
          {isTyping && (
            <div className="typing-indicator">
              <span>COMPASS is typing</span>
              <div className="typing-dots">
                <span></span>
                <span></span>
                <span></span>
              </div>
            </div>
          )}
          
          <div ref={messagesEndRef} />
        </div>
        
        <div className="input-container">
          <TextInput
            value={inputValue}
            onChange={(e) => setInputValue(e.target.value)}
            onKeyPress={(e) => {
              if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                handleSendMessage(inputValue);
              }
            }}
            placeholder="Describe your symptoms..."
            disabled={isTyping}
          />
          <Button
            onClick={() => handleSendMessage(inputValue)}
            disabled={!inputValue.trim() || isTyping}
            variant="primary"
          >
            Send
          </Button>
        </div>
      </Card>
      
      {showEmergencyModal && <EmergencyModal />}
    </div>
  );
};
```

interface DrugInteraction {
  drug1: string;
  drug2: string;
  severity: 'mild' | 'moderate' | 'severe';
  description: string;
  recommendation: string;
}

interface MedicationCheck {
  medication: string;
  dosage: string;
  frequency: string;
}

// Drug interaction database (simplified example)
const INTERACTION_DATABASE: DrugInteraction[] = [
  {
    drug1: 'warfarin',
    drug2: 'aspirin',
    severity: 'severe',
    description: 'Increased risk of bleeding',
    recommendation: 'Avoid combination or monitor INR closely',
  },
  {
    drug1: 'simvastatin',
    drug2: 'amlodipine',
    severity: 'moderate',
    description: 'Increased simvastatin levels',
    recommendation: 'Limit simvastatin to 20mg daily',
  },
  // Add more interactions...
];

export function checkDrugInteractions(
  currentMedications: MedicationCheck[],
  newMedication: MedicationCheck
): DrugInteraction[] {
  const interactions: DrugInteraction[] = [];
  
  currentMedications.forEach((current) => {
    const interaction = findInteraction(
      current.medication.toLowerCase(),
      newMedication.medication.toLowerCase()
    );
    
    if (interaction) {
      interactions.push(interaction);
    }
  });
  
  return interactions;
}

function findInteraction(drug1: string, drug2: string): DrugInteraction | null {
  return INTERACTION_DATABASE.find(
    (interaction) =>
      (interaction.drug1 === drug1 && interaction.drug2 === drug2) ||
      (interaction.drug1 === drug2 && interaction.drug2 === drug1)
  ) || null;
}

export function calculateDrugScore(interactions: DrugInteraction[]): {
  score: number;
  level: 'safe' | 'caution' | 'warning' | 'danger';
  message: string;
} {
  if (interactions.length === 0) {
    return {
      score: 100,
      level: 'safe',
      message: 'No drug interactions detected',
    };
  }
  
  const severityScores = {
    mild: 10,
    moderate: 30,
    severe: 50,
  };
  
  const totalScore = interactions.reduce(
    (sum, interaction) => sum + severityScores[interaction.severity],
    0
  );
  
  const score = Math.max(0, 100 - totalScore);
  
  let level: 'safe' | 'caution' | 'warning' | 'danger';
  let message: string;
  
  if (score >= 80) {
    level = 'safe';
    message = 'Minor interactions only';
  } else if (score >= 60) {
    level = 'caution';
    message = 'Moderate interactions detected';
  } else if (score >= 40) {
    level = 'warning';
    message = 'Significant interactions require review';
  } else {
    level = 'danger';
    message = 'Severe interactions - consult immediately';
  }
  
  return { score, level, message };
}

      await auditLog({

        userId: req.user.id,
        patientId,
        timestamp: new Date(),
        ipAddress: req.ip,
      });
      
      res.json({
        success: true,
        data: patient,
      });
    } catch (error) {
      next(error);
    }
  },

        req.user.id
      );
      
      // Audit log for PHI modification
      await auditLog({
        action: 'PATIENT_UPDATE',
        userId: req.user.id,
        patientId,
        changes: validatedData,
        timestamp: new Date(),
        ipAddress: req.ip,
      });
      
      res.json({
        success: true,

      });
    } catch (error) {
      next(error);
    }
  },

# Request/Response models
class SymptomAnalysisRequest(BaseModel):
    patient_id: str
    message: str
    conversation_history: List[dict]
    current_symptoms: List[str]
    patient_context: dict

class SymptomAnalysisResponse(BaseModel):
    response_message: str
    identified_symptoms: List[str]
    follow_up_questions: List[str]
    urgency_level: str
    confidence_score: float
    next_step: Optional[str]

class ClinicalPredictionRequest(BaseModel):
    patient_data: dict
    symptoms: List[str]
    lab_results: Optional[dict]
    medications: List[str]

class ClinicalPredictionResponse(BaseModel):
    differential_diagnoses: List[dict]
    recommended_tests: List[dict]
    risk_factors: List[dict]
    clinical_guidelines: List[str]

# Health check endpoint
@app.get("/health")
async def health_check():
    return {
        "status": "healthy",
        "timestamp": datetime.utcnow().isoformat(),
        "models_loaded": {
            "symptom_analyzer": symptom_analyzer.is_loaded(),
            "clinical_predictor": clinical_predictor.is_loaded(),
            "medical_nlp": medical_nlp.is_loaded()
        }
    }

# Symptom analysis endpoint
@app.post("/analyze/symptoms", response_model=SymptomAnalysisResponse)
async def analyze_symptoms(request: SymptomAnalysisRequest):
    try:
        logger.info(f"Analyzing symptoms for patient {request.patient_id}")
        
        # Extract medical entities from message
        entities = medical_nlp.extract_entities(request.message)
        
        # Analyze symptoms and generate response
        analysis = symptom_analyzer.analyze(
            message=request.message,
            entities=entities,
            conversation_history=request.conversation_history,
            current_symptoms=request.current_symptoms,
            patient_context=request.patient_context
        )
        
        # Check for emergency conditions
        urgency_level = symptom_analyzer.assess_urgency(
            symptoms=analysis['identified_symptoms'],
            patient_context=request.patient_context
        )
        
        return SymptomAnalysisResponse(
            response_message=analysis['response'],
            identified_symptoms=analysis['identified_symptoms'],
            follow_up_questions=analysis['follow_up_questions'],
            urgency_level=urgency_level,
            confidence_score=analysis['confidence'],
            next_step=analysis.get('next_step')
        )
        
    except Exception as e:
        logger.error(f"Error analyzing symptoms: {str(e)}")
        raise HTTPException(status_code=500, detail=str(e))

# Clinical prediction endpoint
@app.post("/predict/clinical", response_model=ClinicalPredictionResponse)
async def predict_clinical(request: ClinicalPredictionRequest):
    try:
        logger.info("Generating clinical predictions")
        
        # Generate differential diagnoses
        diagnoses = clinical_predictor.generate_differential(
            symptoms=request.symptoms,
            patient_data=request.patient_data,
            lab_results=request.lab_results
        )
        
        # Recommend diagnostic tests
        recommended_tests = clinical_predictor.recommend_tests(
            symptoms=request.symptoms,
            differential_diagnoses=diagnoses,
            patient_data=request.patient_data
        )
        
        # Assess risk factors
        risk_factors = clinical_predictor.assess_risks(
            patient_data=request.patient_data,
            medications=request.medications,
            symptoms=request.symptoms
        )
        
        # Get relevant clinical guidelines
        guidelines = clinical_predictor.get_guidelines(
            diagnoses=diagnoses,
            patient_data=request.patient_data
        )
        
        return ClinicalPredictionResponse(
            differential_diagnoses=diagnoses,
            recommended_tests=recommended_tests,
            risk_factors=risk_factors,
            clinical_guidelines=guidelines
        )
        
    except Exception as e:
        logger.error(f"Error generating clinical predictions: {str(e)}")
        raise HTTPException(status_code=500, detail=str(e))

## Initial Setup

### 1. Clone the Repository

```bash
git clone https://github.com/your-org/attending-medical-ai.git
cd attending-medical-ai
```

### 2. Install Dependencies

```bash
# Install root dependencies
npm install

# Bootstrap all packages and apps
npm run bootstrap
```

### 3. Environment Configuration

```bash
# Copy example environment file
cp .env.example .env

# Edit .env with your configuration
# Required: DATABASE_URL, REDIS_URL, JWT_SECRET
```

### 4. Database Setup

```bash
# Using Docker
docker-compose up postgres redis

# Or manually with PostgreSQL
createdb attending_medical
npm run db:migrate
npm run db:seed
```

## Development Workflow

### Running Services

```bash
# Start all services (recommended)
docker-compose up

# Or run individually
npm run dev:patient-portal    # http://localhost:3001
npm run dev:provider-portal   # http://localhost:3002
npm run dev:backend          # http://localhost:3000
npm run dev:ai-service       # http://localhost:8000
```

### Working with the Monorepo

```bash
# Run command in specific workspace
npm run dev --workspace=@attending/patient-portal

# Add dependency to specific package
npm install axios --workspace=@attending/api-client

# Build all packages
npm run build

# Run tests
npm test
npm run test:medical  # Medical-specific tests
```

## Code Standards

### TypeScript
- Strict mode enabled
- All code must be properly typed
- No `any` types without justification

### Medical Code Standards
- All medical calculations must have tests
- Drug interactions must be validated
- PHI handling must follow HIPAA guidelines
- Emergency protocols must be clearly marked

### Testing
- Unit tests for all utilities
- Integration tests for API endpoints
- E2E tests for critical user flows
- Medical accuracy validation tests

## Common Tasks

### Adding a New Component

```bash
# Create component in ui-system
cd packages/ui-system/src/components
mkdir MyComponent
touch MyComponent/MyComponent.tsx
touch MyComponent/MyComponent.test.tsx
touch MyComponent/MyComponent.stories.tsx
touch MyComponent/index.ts
```

### Adding a New API Endpoint

1. Define types in `packages/types`
2. Add controller in `services/backend/src/controllers`
3. Add service logic in `services/backend/src/services`
4. Add route in `services/backend/src/routes`
5. Add client method in `packages/api-client`
6. Write tests

### Database Migrations

```bash
# Create new migration
npm run db:migration:create -- --name add_patient_notes

# Run migrations
npm run db:migrate

# Rollback
npm run db:rollback
```

## Troubleshooting

### Common Issues

1. **Port already in use**
   ```bash
   # Find and kill process
   lsof -ti:3000 | xargs kill -9
   ```

2. **Database connection errors**
   ```bash
   # Check PostgreSQL is running
   docker-compose ps
   # Restart database
   docker-compose restart postgres
   ```

3. **Package linking issues**
   ```bash
   # Clean and reinstall
   npm run clean
   npm install
   npm run bootstrap
   ```
